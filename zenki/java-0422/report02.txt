ソース YokinApp2.java

class YokinApp2{
    public static void main(String args[]){
	Yokin myyokin;
	myyokin = new Yokin();

	int mygankin = Integer.parseInt(args[0]);

	//myyokin.gankin = 100000;
	myyokin.gankin = mygankin;

	double kotae;
	kotae = myyokin.risoku(3);
	System.out.println ("利息は" + kotae);

	System.exit(0);
    }
}

実行

[p160101@java23 java-0422]$ java YokinApp2 10000
利息は5.0
[p160101@java23 java-0422]$ java YokinApp2 20000
利息は10.0
