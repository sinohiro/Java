1.FlowLayoutの練習
ソースコード: FlowLayoutTest.java
import java.awt.*;
import java.awt.event.*;
import javax.swing.*;


public class FlowLayoutTest extends JFrame implements ActionListener {
	private JButton button;
	private JButton button2;
	private JLabel	countlabel;
	private int			count = 1;
	private int			count2 = 1;

	public FlowLayoutTest() {
		super();
		this.button = new JButton("今日に比べて");
		this.button2 = new JButton("昨日に比べて");
		JLabel label1 = new JLabel("今回のプロジェクト");
		JPanel panel1 = new JPanel();
		panel1.setLayout(new FlowLayout());
		panel1.add(label1);
		panel1.add(this.button);
		panel1.add(this.button2);
		getContentPane().add(panel1);
		this.button.addActionListener(this);
		this.button2.addActionListener(this);
	}

	public void actionPerformed(ActionEvent e) {
		if (e.getSource() == this.button) {
			count++;
			this.button.setText("進捗どうですか? X " + count);
			System.out.println(count);
		}
		if (e.getSource() == this.button2) {
			count2++;
			this.button2.setText("進捗どうですか? X " + count2);
			System.out.println(count2);
		}

	}


	public static void main(String[] args) {
		JFrame mainFrame = new JFrame("4J_21_shinodaApp");
		mainFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		FlowLayoutTest a = new FlowLayoutTest();
		a.setSize(640, 480);
		a.setLocation(100, 100);
		a.setVisible(true);
		a.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
		a.addWindowListener(new WindowAdapter(){
			public void windowClosing(WindowEvent e){
				System.exit(0);
			}
		});
	}
}


開発者向け解説
FlowLayoutでレイアウトを１つ作りその中にラベルとボタン２つを入れて表示しています

開発者向け解説
BorderLayoutでNORSHにラベル、WEST,EASTにそれぞれボタンを一つ入れて表示しています

課題1との違い
FlowLayoutは横にボタンやラベルを並べる際に便利なレイアウトでBorderLayoutは上下左右にボタンやレイアウトを配置するときに便利なレイアウトです
どちらのレイアウトにも利点があり状況によってうまく使い分ける必要があると思いました.

3.複雑なウインドウを作成
ソースコード:ComplexFrame.java
import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class ComplexFrame extends JFrame implements ActionListener {
	private JButton button;
	private JButton button2;
	private JButton button3;
	private JLabel	countlabel;
	private int			count = 1;
	private int			count2 = 1;
	private int			count3 = 1;

	public ComplexFrame() {
		super();
		this.button = new JButton("今日に比べて");
		this.button2 = new JButton("昨日に比べて");
		this.button3 = new JButton("救ってほしいですか?");
		JLabel label1 = new JLabel("今回のプロジェクト");

		JPanel panel1 = new JPanel();
		panel1.setLayout(new FlowLayout());
		panel1.add(label1);
		panel1.add(this.button);
		panel1.add(this.button2);

		JPanel panel2 = new JPanel();
		panel2.setLayout(new BorderLayout());
		panel2.add(panel1, BorderLayout.CENTER);
		panel2.add(button3, BorderLayout.NORTH);

		getContentPane().add(panel2);
		this.button.addActionListener(this);
		this.button2.addActionListener(this);
	}

	public void actionPerformed(ActionEvent e) {
		if (e.getSource() == this.button) {
			count++;
			this.button.setText("進捗どうですか? X " + count);
			System.out.println(count);
		}
		if (e.getSource() == this.button2) {
			count2++;
			this.button2.setText("進捗どうですか? X " + count2);
			System.out.println(count2);
		}
		if (e.getSource() == this.button3) {
			count3++;
			this.button2.setText("進捗どうですか? X " + count3);
			System.out.println(count3);
		}
	}

	public static void main(String[] args) {
		JFrame mainFrame = new JFrame("4J_21_shinodaApp");
		mainFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		ComplexFrame a = new ComplexFrame();
		a.setSize(640, 480);
		a.setLocation(100, 100);
		a.setVisible(true);
		a.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
		a.addWindowListener(new WindowAdapter(){
			public void windowClosing(WindowEvent e){
				System.exit(0);
			}
		});
	}
}

開発者向け解説をレポート
panel1をFlowLayoutで作成しその中にラベル、ボタンを配置します
そのpanel1をBorderLayoutで作成したpanel2の中にCENTERで表示し、
同時にもう一つボタンを作りそのボタンをNORTHに配置しました
